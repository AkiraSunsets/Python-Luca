# 1) Crie uma função chamada "mostrar_hora" que imprime a hora atual do sistema.
from datetime import datetime

def mostrar_hora():
    hora_atual = datetime.now()
    hora_formatada = hora_atual.strftime("%H:%M:%S")
    print(hora_formatada)
mostrar_hora()

# 4) Escreva uma função chamada "imprimir_triangulo" que imprime um triângulo formado por asteriscos.
def imprimir_triangulo(linhas):
    for i in range(1, linhas + 1):
        for j in range(linhas - i):
            print("", end="")
            for j in range(2 * i - 1):
                print("*", end="")
            print()
imprimir_triangulo(5)

# Desafio) Foi realizada uma pesquisa de algumas características e gostos de quatro habitantes incluindo:
# nome, sexo, esporte favorito (Natação, Futebol, Volêi, Tênis) e idade. Com esses dados faça:
# - Função que armazene os dados em uma lista. Dica: Use dicionários dentro da lista.
# - Calcule a idade média de homens que gostam de natação. Caso não haja homens que gostam de natação chame uma função e
# imprima um aviso de que não há homens que gostam de natação.

def aviso_sem_homens_natacao():
    print("Não há homens que gostam de natação.")

def obter_habitantes():
    return [
        {"nome": "João", "sexo": "M", "esporte": "Natação", "idade": 25},
        {"nome": "Maria", "sexo": "F", "esporte": "Vôlei", "idade": 22},
        {"nome": "Carlos", "sexo": "M", "esporte": "Futebol", "idade": 30},
        {"nome": "Ana", "sexo": "F", "esporte": "Tênis", "idade": 28}
    ]

def calcular_media_idade_natacao(habitantes):
    homens_natacao = [h for h in habitantes if h["sexo"] == "M" and h["esporte"] == "Natação"]

    if not homens_natacao:
        aviso_sem_homens_natacao()
    else:
        media = sum(h["idade"] for h in homens_natacao) / len(homens_natacao)
        print(f"Média de idade dos homens que gostam de natação: {media:.2f}")

habitantes = obter_habitantes()
calcular_media_idade_natacao(habitantes)
